================================================================================
boolean/float/integer/byte/char literal string interpolation
================================================================================
fn main { 
  ignore("\{42}")
  ignore("\{false}")
  ignore("\{'f'}")
  ignore("\{b'\xff'}")
  ignore("\{b'a'}")
  ignore("\{0.3}")
}
--------------------------------------------------------------------------------


================================================================================
Binary expr string interpolation
================================================================================
fn main { 
  ignore("\{41 + 1}")
  ignore("\{false && true || true && false}")
}
--------------------------------------------------------------------------------


================================================================================
Qualified identifier string interpolation
================================================================================
fn main { 
  ignore("\{@pkg.foo}")
  ignore("\{@immut/list.from_array([0, 2, 3])}")
}
--------------------------------------------------------------------------------


================================================================================
Unary expression string interpolation
================================================================================
fn main { 
  ignore("\{+42}")
  ignore("\{-42}")
}
--------------------------------------------------------------------------------


================================================================================
Apply expression string interpolation
================================================================================
fn main {
  ignore("\{f(1, g?(3))}")
  ignore("\{f(1, g?())}")
}
--------------------------------------------------------------------------------
